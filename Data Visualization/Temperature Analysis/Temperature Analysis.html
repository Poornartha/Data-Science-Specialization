Note: The data for this assignment comes from a subset of The National Centers for Environmental Information (NCEI) Daily Global Historical Climatology Network (GHCN-Daily). The GHCN-Daily is comprised of daily climate records from thousands of land surface stations across the globe.

Each row in the assignment datafile corresponds to a single observation.

The following variables are provided to you:

  • id : station identification code
  • date : date in YYYY-MM-DD format (e.g. 2012-01-24 = January 24, 2012)
  • element : indicator of element type
    • TMAX : Maximum temperature (tenths of degrees C)
    • TMIN : Minimum temperature (tenths of degrees C)
  • value : data value for element (tenths of degrees C)

import matplotlib.pyplot as plt
import mplleaflet
import pandas as pd
import numpy as np
import matplotlib.dates as dates

# import dataset
dataset = 'data/C2A2_data/BinnedCsvs_d400/fb441e62df2d58994928907a91895ec62c2c42e6cd075c2700843b89.csv'
df = pd.read_csv(dataset)

# convert temp to correct Celcius values, NOAA using tenths of degree celcius
df = df.sort_values(by=['Data_Value'])
df['Data_Value'] *= 1/10 

# remove 29 Feb entries from dataset
df = df.sort_values(by=['Date'])
df['Year'] = pd.DatetimeIndex(df['Date']).year
df['Month'] = pd.DatetimeIndex(df['Date']).month
df['Day'] = pd.DatetimeIndex(df['Date']).day

# format month and day to double digits
df['Month'] = df['Month'].map("{:02}".format)
df['Day'] = df['Day'].map("{:02}".format)

# create month-date column
df['Month-Date'] = df['Month'].astype(str) + "-" + df['Day'].astype(str)
df = df[df['Month-Date'] != '02-29']

# split 2015 dataset from non2015 dataset (2004-2014)
df_2015 = df[df['Year'] == 2015]
df_non2015 = df[df['Year'] != 2015]

# find the min and max temps for each dataset and group by date
min_2015 = df_2015[df_2015['Element'] == 'TMIN'].groupby('Month-Date').agg({'Data_Value': np.min})
max_2015 = df_2015[df_2015['Element'] == 'TMAX'].groupby('Month-Date').agg({'Data_Value': np.max})

min_non2015 = df_non2015[df_non2015['Element'] == 'TMIN'].groupby('Month-Date').agg({'Data_Value': np.min})
max_non2015 = df_non2015[df_non2015['Element'] == 'TMAX'].groupby('Month-Date').agg({'Data_Value': np.max})

# find days in 2015 that broke record low and high from 2004-2014
record_min = np.where(min_2015 < min_non2015)[0]
record_max = np.where(max_2015 > max_non2015)[0]

plt.figure()

# plot the min and max records from 2004-2014 and fill range between max and low temps
plt.plot(min_non2015.values, 'b', lw= 0.5, label = 'Record Low')
plt.plot(max_non2015.values, 'r', lw= 0.5, label = 'Record High')
plt.fill_between(range(len(min_non2015)), min_non2015['Data_Value'], max_non2015['Data_Value'], facecolor = 'grey', alpha = 0.25)

# overlay 2015 data that broke record in scatterplot form
plt.scatter(record_min, min_2015.iloc[record_min], s = 25, marker='o', c = 'green', label = '2015 Low')
plt.scatter(record_max, max_2015.iloc[record_max], s = 25, marker='o', c = 'black', label = '2015 High')

# add labels to axes
plt.xlabel('Month-Day')
plt.xticks(range(0, len(min_non2015), 20), min_non2015.index[range(0, len(min_non2015), 20)], rotation = '90')

plt.ylabel('Temperature in Celsius')
plt.title('2015 Record days vs 2005-2014 data for Ann Arbor, Michigan')

# add legend
plt.legend(loc='best')

# remove top and right spines
ax = plt.gca()
ax.spines['right'].set_visible(False)
ax.spines['top'].set_visible(False)

plt.show()
